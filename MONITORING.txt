        // Switch Tabs
        function switchTab(tabName) {
            // Update nav tabs
            document.querySelectorAll('.nav-tab').forEach(tab => {
                tab.classList.remove('active');
            });
            event.target.classList.add('active');

            // Update content sections
            document.querySelectorAll('.content-section').forEach(section => {
                section.classList.remove('active');
            });
            document.getElementById(`${tabName}-section`).classList.add('active');

            // Load content for specific tabs
            if (tabName === 'equipment') {
                loadEquipmentSection();
            } else if (tabName === 'maintenance') {
                loadMaintenanceSection();
            } else if (tabName === 'alerts') {
                loadAlertsSection();
            }
        }

        // Search Functionality
        document.getElementById('searchInput').addEventListener('input', function(e) {
            const searchTerm = e.target.value.toLowerCase();
            const cards = document.querySelectorAll('.classroom-card');
            
            cards.forEach(card => {
                const text = card.textContent.toLowerCase();
                if (text.includes(searchTerm)) {
                    card.style.display = 'block';
                } else {
                    card.style.display = 'none';
                }
            });
        });

        // Filter Functions
        function filterByBlock(block) {
            const cards = document.querySelectorAll('.classroom-card');
            
            cards.forEach(card => {
                if (block === 'all') {
                    card.style.display = 'block';
                } else {
                    const cardText = card.textContent;
                    if (cardText.includes(`Block ${block}`)) {
                        card.style.display = 'block';
                    } else {
                        card.style.display = 'none';
                    }
                }
            });
            
            // Update button states
            document.querySelectorAll('.filter-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');
        }

        function filterByStatus(status) {
            const cards = document.querySelectorAll('.classroom-card');
            
            cards.forEach(card => {
                const statusBadge = card.querySelector('.status-badge');
                if (statusBadge) {
                    const cardStatus = statusBadge.textContent.toLowerCase().trim();
                    if (cardStatus === status) {
                        card.style.display = 'block';
                    } else {
                        card.style.display = 'none';
                    }
                }
            });
            
            // Update button states
            document.querySelectorAll('.filter-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');
        }

        // Update Statistics
        function updateStats() {
            const available = classrooms.filter(c => c.status === 'available').length;
            const occupied = classrooms.filter(c => c.status === 'occupied').length;
            
            document.getElementById('availableRooms').textContent = available;
            document.getElementById('occupiedRooms').textContent = occupied;
        }

        // Show Alert
        function showAlert(message, type) {
            const alertContainer = document.getElementById('alertContainer');
            const alert = document.createElement('div');
            alert.className = `alert alert-${type}`;
            alert.innerHTML = `
                <div class="alert-icon">
                    <i class="fas fa-${type === 'success' ? 'check-circle' : 'exclamation-triangle'}"></i>
                </div>
                <div class="alert-content">
                    <div class="alert-title">${type === 'success' ? 'Success' : 'Alert'}</div>
                    <div class="alert-message">${message}</div>
                </div>
            `;
            
            alertContainer.appendChild(alert);
            
            setTimeout(() => {
                alert.style.animation = 'slideOutRight 0.5s ease-out';
                setTimeout(() => {
                    alert.remove();
                }, 500);
            }, 3000);
        }

        // Check Maintenance Alerts
        function checkMaintenanceAlerts() {
            const today = new Date();
            equipment.forEach(eq => {
                const nextServiceDate = new Date(eq.projector.nextService);
                const daysDiff = Math.floor((nextServiceDate - today) / (1000 * 60 * 60 * 24));
                
                if (daysDiff <= 30 && daysDiff > 0) {
                    const classroom = classrooms.find(c => c.id === eq.classroomId);
                    showAlert(`Projector service due for ${classroom.name} in ${daysDiff} days`, 'warning');
                }
            });
        }

        // Local Storage Functions
        function saveDataToStorage() {
            localStorage.setItem('classrooms', JSON.stringify(classrooms));
            localStorage.setItem('equipment', JSON.stringify(equipment));
        }

        function loadDataFromStorage() {
            const storedClassrooms = localStorage.getItem('classrooms');
            const storedEquipment = localStorage.getItem('equipment');
            
            if (storedClassrooms) {
                const data = JSON.parse(storedClassrooms);
                data.forEach((stored, index) => {
                    if (classrooms[index]) {
                        classrooms[index].status = stored.status;
                        classrooms[index].benches = stored.benches;
                        classrooms[index].projector = stored.projector;
                        classrooms[index].cpu = stored.cpu;
                    }
                });
                renderClassrooms();
                updateStats();
            }
        }
